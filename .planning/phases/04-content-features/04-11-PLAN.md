    Author `src/components/public/used-in-section.tsx`:
    ```typescript
    import { getUsedInForProduct } from '@/lib/used-in';
    import { RecipeCard } from './recipe-card';
    import { IndustryCard } from './industry-card';
    import { getTranslations } from 'next-intl/server';

    interface Props {
      productId: string;
      locale: 'uz' | 'ru' | 'en';
    }

    export async function UsedInSection({ productId, locale }: Props) {
      const { recipes, industries } = await getUsedInForProduct(productId, locale);
      if (recipes.length === 0 && industries.length === 0) return null;
      const t = await getTranslations({ locale, namespace: 'UsedIn' });
      return (
        <section aria-labelledby="used-in-heading" className="...">
          <h2 id="used-in-heading">{t('title')}</h2>
          {recipes.length > 0 && (
            <div className="...">
              <h3>{t('recipes')}</h3>
              <ul className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
                {recipes.map((r) => (
                  <li key={r.id}>
                    <RecipeCard recipe={{ slug: r.slug, title: r.title, excerpt: r.excerpt, featuredImagePublicId: r.featuredImagePublicId }} locale={locale} />
                  </li>
                ))}
              </ul>
            </div>
          )}
          {industries.length > 0 && (
            <div className="...">
              <h3>{t('industries')}</h3>
              <ul className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
                {industries.map((i) => (
                  <li key={i.id}>
                    <IndustryCard industry={{ slug: i.slug, title: i.title, excerpt: i.excerpt, featuredImagePublicId: i.featuredImagePublicId }} locale={locale} />
                  </li>
                ))}
              </ul>
            </div>
          )}
        </section>
      );
    }
    ```

    Add translations to messages/{uz,ru,en}.json:

    - UsedIn.recipes = "Примеры применения" / "Foydalanish misollari" / "Application examples"
    - UsedIn.industries = "Отрасли" / "Sohalar" / "Industries"

    The cap-at-6 is enforced by getUsedInForProduct's `.slice(0, 6)` per type (verified in 04-03 spec 1).
