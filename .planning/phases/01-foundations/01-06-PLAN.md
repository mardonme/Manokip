### middleware.ts (RESEARCH.md Pattern 2, lines 341–370):
```ts
import createMiddleware from 'next-intl/middleware';
import NextAuth from 'next-auth';
import authConfig from '@/lib/auth.config';
import { routing } from '@/i18n/routing';

const handleI18nRouting = createMiddleware(routing);
const { auth } = NextAuth(authConfig);

export default auth(async function middleware(req) {
  const pathname = req.nextUrl.pathname;

  // Admin gate: requires valid Auth.js session (T-ADMIN-GATE, D-15)
  const isAdminPath = /^\/(uz|ru|en)\/admin(\/|$)/.test(pathname);
  if (isAdminPath && !req.auth) {
    const locale = pathname.split('/')[1] || 'uz';
    const url = req.nextUrl.clone();
    url.pathname = `/${locale}/login`;
    return Response.redirect(url, 307);
  }

  // Locale redirect + detection chain (D-03: cookie → Accept-Language → uz)
  return handleI18nRouting(req);
});

export const config = {
  matcher: [
    // Skip: api routes, Next internals, static assets (anything with a file extension)
    '/((?!api|_next/static|_next/image|_vercel|favicon.ico|robots.txt|sitemap.xml|.*\\..*).*)',
  ],
};
```
**CRITICAL (Pitfall 1):** The only static import from `src/lib/*` is `auth.config.ts`. Do NOT add `from '@/lib/auth'` — that pulls the Drizzle Neon TCP driver into Edge and crashes at build.

### src/app/api/cloudinary/sign/route.ts (RESEARCH.md Pattern 8, lines 632–672):
```ts
import { cloudinary } from '@/lib/cloudinary';
import { auth } from '@/lib/auth';
import { env } from '@/env';
import { z } from 'zod';

const FOLDER_ALLOWLIST = ['products', 'recipes', 'industries', 'manufacturers'] as const;

const bodySchema = z.object({
  folder: z.enum(FOLDER_ALLOWLIST),
});

export async function POST(req: Request) {
  // T-CLD-01: admin-session gate
  const session = await auth();
  if (!session?.user?.email) {
    return new Response('Unauthorized', { status: 401 });
  }

  // T-CLD-02: folder allowlist
  let raw: unknown;
  try {
    raw = await req.json();
  } catch {
    return new Response('Invalid JSON', { status: 400 });
  }
  const parsed = bodySchema.safeParse(raw);
  if (!parsed.success) {
    return new Response('Invalid folder', { status: 400 });
  }
  const { folder } = parsed.data;

  // T-CLD-03: signature TTL — Cloudinary signatures are valid 1h server-side; we tag timestamp
  // with Math.floor(Date.now()/1000) and the client sends the same integer in its upload
  // form so HMAC matches (Pitfall 5).
  const timestamp = Math.floor(Date.now() / 1000);
  const signature = cloudinary.utils.api_sign_request(
    { timestamp, folder },
    env.CLOUDINARY_API_SECRET,
  );

  return Response.json({
    signature,
    timestamp,
    folder,
    apiKey: env.CLOUDINARY_API_KEY,
    cloudName: env.CLOUDINARY_CLOUD_NAME,
  });
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Edge middleware → Node-only modules | Middleware MUST NOT import DB / adapter / Node-net modules — crashes at runtime |
| untrusted client → /api/cloudinary/sign | Unauthenticated POST accepts folder string; must reject without session + reject out-of-allowlist folder |
| client uploader → Cloudinary | Signed credentials travel to the browser; 15-min effective window via timestamp |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-LOC-01 | Tampering (locale-param injection) | `middleware.ts` | mitigate | Admin-path regex is locked to `/^\/(uz\|ru\|en)\/admin(\/\|$)/` — no interpolation of user input into the redirect `pathname`. `next-intl createMiddleware` enforces the same allowlist for the locale redirect. Verified by `tests/e2e/locale-redirect.spec.ts` (plan 04, /xx/ 404 test). |
| T-ADMIN-GATE | Elevation of Privilege (unauth access to /admin) | `middleware.ts` | mitigate | `isAdminPath && !req.auth` triggers 307 redirect to `/{locale}/login`. `req.auth` comes from Auth.js `auth()` wrapper reading the signed session cookie — cryptographic verification at Edge. Authoritative DB session read happens in `requireAdmin()` (Node, in the admin page) as defense-in-depth. Verified by `tests/e2e/admin-gate.spec.ts`. |
| T-AUTH-01 | Spoofing (magic-link replay) | `/api/auth/[...nextauth]` (inherited from plan 05) | mitigate | Auth.js single-use verification tokens; `tests/e2e/magic-link-login.spec.ts` drives the round-trip. |
| T-AUTH-02 | Elevation of Privilege | `signIn` callback (from plan 05) | mitigate | Covered by `tests/db/auth-signin-callback.test.ts` (plan 05). |
| T-CLD-01 | Elevation of Privilege (unauth signing) | `/api/cloudinary/sign` | mitigate | `await auth()` returns null without session → 401 early exit BEFORE reading body. Verified by `tests/api/cloudinary-sign.test.ts` case "401 no session". |
| T-CLD-02 | Tampering (folder traversal / unauth folder writes) | `/api/cloudinary/sign` | mitigate | Zod enum `['products','recipes','industries','manufacturers']` rejects anything else with 400. Verified by `tests/api/cloudinary-sign.test.ts` case "400 bad folder". |
| T-CLD-03 | TTL abuse | Cloudinary signature | mitigate | `timestamp = Math.floor(Date.now()/1000)` — integer seconds. Cloudinary rejects signatures where `timestamp` drifts > 1h from server time, giving an effective 15-min window because client upload must happen shortly after the sign request. |
| T-SEC-ENV | Information Disclosure | Sign endpoint response | mitigate | Response returns `apiKey` + `cloudName` (public), NEVER `apiSecret` (server-only). |
