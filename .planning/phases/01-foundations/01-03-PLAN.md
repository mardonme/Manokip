<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| developer machine → Neon direct URL | `drizzle-kit migrate` authenticates with `DATABASE_URL_DIRECT` (stronger privilege than runtime pooled URL) |
| Vercel build runner → Neon direct URL | `vercel.json` buildCommand runs migrate with env secret |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-LOC-01 | Tampering | Live DB `*_translations` CHECK | mitigate | Migration creates CHECK constraints at the DB layer; `tests/db/locale-constraint.test.ts` (Task 03.3) confirms a rogue `locale='de'` INSERT is rejected by Postgres, not just by app code. |
| T-SEC-ENV | Information Disclosure | `vercel.json` | mitigate | Uses `@database-url-direct` secret reference (not literal); no credential committed. |
| T-MIGRATE-01 | Denial of Service | `vercel.json` buildCommand | accept | If migration fails during a production deploy, Vercel blocks the deploy (desired behavior — don't serve traffic against an out-of-date schema). Failing loud > failing silently. |
</threat_model>

    2. **tests/db/locale-constraint.test.ts** — asserts live DB rejects `locale='de'`:

       import { describe, it, expect } from 'vitest';
       import { sql } from 'drizzle-orm';
       import { getTestDb, requireTestDatabaseUrl } from '../_fixtures/db';

       describe('FOUND-01: product_translations CHECK rejects invalid locale', () => {
         it('rejects locale=de with CHECK violation', async () => {
           requireTestDatabaseUrl();
           const db = await getTestDb();
           // Create a product to reference
           const [product] = await db.execute(sql`
             INSERT INTO category (id) VALUES (gen_random_uuid()) RETURNING id
           `).then((r) => r.rows as Array<{ id: string }>);
           // product requires category_id; create both in a transaction-like sequence
           const catId = product.id;
           const prodRes = await db.execute(sql`
             INSERT INTO product (category_id) VALUES (${catId}::uuid) RETURNING id
           `);
           const prodId = (prodRes.rows as Array<{ id: string }>)[0].id;

           // Now attempt the invalid-locale insert
           await expect(
             db.execute(sql`
               INSERT INTO product_translations (product_id, locale, name, slug)
               VALUES (${prodId}::uuid, 'de', 'Test', 'test-${Date.now()}')
             `),
           ).rejects.toThrow(/product_translations_locale_check|check constraint/i);

           // Cleanup
           await db.execute(sql`DELETE FROM product WHERE id = ${prodId}::uuid`);
           await db.execute(sql`DELETE FROM category WHERE id = ${catId}::uuid`);
         });

         it('accepts locale=uz + ru + en', async () => {
           requireTestDatabaseUrl();
           const db = await getTestDb();
           const catRes = await db.execute(sql`INSERT INTO category (id) VALUES (gen_random_uuid()) RETURNING id`);
           const catId = (catRes.rows as Array<{ id: string }>)[0].id;
           const prodRes = await db.execute(sql`INSERT INTO product (category_id) VALUES (${catId}::uuid) RETURNING id`);

       import { describe, it, expect } from 'vitest';
       import { sql } from 'drizzle-orm';
       import { getTestDb, requireTestDatabaseUrl } from '../_fixtures/db';

       describe('FOUND-02: product_spec_values typed insert + range query', () => {
         it('inserts num_value=42.5 and range query returns the row', async () => {
           requireTestDatabaseUrl();
           const db = await getTestDb();

           const catRes = await db.execute(sql`INSERT INTO category (id) VALUES (gen_random_uuid()) RETURNING id`);
           const catId = (catRes.rows as Array<{ id: string }>)[0].id;
           const specRes = await db.execute(sql`
             INSERT INTO spec_field (category_id, key, data_type, unit, filter_kind)
             VALUES (${catId}::uuid, 'pressure_max', 'number', 'bar', 'range')
             RETURNING id
           `);
           const specId = (specRes.rows as Array<{ id: string }>)[0].id;
           const prodRes = await db.execute(sql`INSERT INTO product (category_id) VALUES (${catId}::uuid) RETURNING id`);
           const prodId = (prodRes.rows as Array<{ id: string }>)[0].id;

           // Typed num_value insert (NOT opaque value TEXT)
           await db.execute(sql`
             INSERT INTO product_spec_values (product_id, spec_field_id, num_value, unit)
             VALUES (${prodId}::uuid, ${specId}::uuid, 42.5, 'bar')
           `);

           // Range query — pressure >= 40 AND pressure <= 50
           const result = await db.execute(sql`
             SELECT id, num_value::text AS num_value
             FROM product_spec_values
             WHERE spec_field_id = ${specId}::uuid
               AND num_value >= 40 AND num_value <= 50
           `);
           expect(result.rows.length).toBe(1);
           expect(parseFloat((result.rows[0] as { num_value: string }).num_value)).toBe(42.5);

           // Cleanup
           await db.execute(sql`DELETE FROM product_spec_values WHERE product_id = ${prodId}::uuid`);
           await db.execute(sql`DELETE FROM product WHERE id = ${prodId}::uuid`);
           await db.execute(sql`DELETE FROM spec_field WHERE id = ${specId}::uuid`);
           await db.execute(sql`DELETE FROM category WHERE id = ${catId}::uuid`);
         });
