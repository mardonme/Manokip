### src/lib/auth.ts (Node-runtime — RESEARCH.md lines 508–548 + D-09 enforcement)
```ts
import NextAuth from 'next-auth';
import { DrizzleAdapter } from '@auth/drizzle-adapter';
import { eq, and, isNull } from 'drizzle-orm';
import { db } from '@/db/client';
import { authUsers, authAccounts, sessions, verificationTokens, adminUsers } from '@/db/schema';
import authConfig from './auth.config';

const ABSOLUTE_SESSION_SECONDS = 7 * 24 * 60 * 60;  // D-09 7d absolute cap
const IDLE_SESSION_SECONDS     = 24 * 60 * 60;      // D-09 24h idle cap (sliding — Auth.js session.maxAge)

export const { handlers, auth, signIn, signOut } = NextAuth({
  ...authConfig,
  adapter: DrizzleAdapter(db, {
    usersTable: authUsers,
    accountsTable: authAccounts,
    sessionsTable: sessions,
    verificationTokensTable: verificationTokens,
  }),
  // D-09: 24h idle handled by Auth.js sliding sessions (maxAge sets sessions.expires on each
  // authenticated request the adapter sees). 7d absolute cap handled by requireAdmin() via the
  // absoluteExpires column (populated in the session callback below, stamped once via isNull guard).
  session: { strategy: 'database', maxAge: IDLE_SESSION_SECONDS, updateAge: 60 * 60 },
  callbacks: {
    async signIn({ user }) {                // D-10 + T-AUTH-02
      if (!user?.email) return false;
      const adminUser = await db
        .select({ email: adminUsers.email, role: adminUsers.role, active: adminUsers.active })
        .from(adminUsers)
        .where(and(eq(adminUsers.email, user.email), eq(adminUsers.active, true)))
        .limit(1)
        .then((rows) => rows[0]);
      if (!adminUser || adminUser.role !== 'admin') return false;  // D-11: v1 only 'admin'
      return true;
    },
    async session({ session }) {
      // D-09: lazily stamp absoluteExpires the first time a session is read without one set.
      // Stamp-once: isNull guard ensures subsequent reads are a no-op UPDATE (Postgres optimizes
      // a zero-row UPDATE). For newly-created sessions (via DrizzleAdapter), absoluteExpires is
      // NULL on creation → first session() call stamps it to now()+7d.
      if (session?.sessionToken) {
        await db
          .update(sessions)
          .set({ absoluteExpires: new Date(Date.now() + ABSOLUTE_SESSION_SECONDS * 1000) })
          .where(and(
            eq(sessions.sessionToken, session.sessionToken),
            isNull(sessions.absoluteExpires),
          ));
      }
      return session;
    },
  },
});

export async function requireAdmin() {
  const session = await auth();
  if (!session?.user?.email) {
    throw new Error('Unauthorized');
  }

  // D-09 enforcement: reject sessions past their 7d absolute cap.
  // (24h idle is enforced by Auth.js sliding sessions via session.maxAge above — sessions.expires
  // < now() causes auth() to return null, which the !session check above already rejects.)
  if (session.sessionToken) {
    const [row] = await db
      .select({ absoluteExpires: sessions.absoluteExpires })
      .from(sessions)
      .where(eq(sessions.sessionToken, session.sessionToken))
      .limit(1);
    if (row?.absoluteExpires && row.absoluteExpires.getTime() < Date.now()) {
      // Session is past its 7d absolute cap — delete it and reject.
      await db.delete(sessions).where(eq(sessions.sessionToken, session.sessionToken));
      throw new Error('Unauthorized');  // D-09 absolute timeout
    }
  }

  return session;
}
```

### src/lib/bootstrap.ts (RESEARCH.md lines 906–920 — D-12, Pitfall 8)
```ts
import { db } from '@/db/client';
import { adminUsers } from '@/db/schema';
import { env } from '@/env';

let bootstrapped = false;

export async function bootstrapAdmin() {
  if (bootstrapped) return;
  const email = env.BOOTSTRAP_ADMIN_EMAIL;
  if (!email) { bootstrapped = true; return; }

  // D-12 verbatim enforcement: "if admin_user is empty and the env var is set, insert one row"
  // Pre-check: if ANY admin already exists, bootstrap is a no-op — do not insert, even if the
  // email is new. This protects against a stale BOOTSTRAP_ADMIN_EMAIL being changed to a
  // different address after real admins have been added (the ON CONFLICT only covers same-email
  // conflict, not "a different admin already exists").
  const [existing] = await db
    .select({ email: adminUsers.email })
    .from(adminUsers)
    .limit(1);
  if (existing) {
    bootstrapped = true;
    return;
  }

  // Belt-and-suspenders: onConflictDoNothing guards against concurrent cold starts racing the
  // same insert (Pitfall 8). The pre-check above is the authoritative D-12 guard; this is defense-in-depth.
  await db
    .insert(adminUsers)
    .values({ email, role: 'admin', active: true })
    .onConflictDoNothing();

  bootstrapped = true;
}
```
Called from `src/instrumentation.ts` (created in plan 07) under `NEXT_RUNTIME === 'nodejs'`, so it runs once per cold-start.

```ts
Resend({
  apiKey: process.env.AUTH_RESEND_KEY,
  from: process.env.RESEND_FROM_EMAIL,
  async sendVerificationRequest({ identifier: to, url, provider }) {
    const { Resend: ResendSDK } = await import('resend');
    const { render } = await import('@react-email/render');
    const MagicLinkEmail = (await import('@/emails/magic-link')).default;
    const resend = new ResendSDK(provider.apiKey as string);
    const html = await render(MagicLinkEmail({ url, locale: 'uz' }));
    const { error } = await resend.emails.send({
      from: provider.from as string,
      to,
      subject: 'Manometr — kirish havolasi',
      html,
    });
    if (error) throw new Error(error.message);
  },
}),
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-AUTH-01 | Spoofing (magic-link token replay) | Auth.js `verification_tokens` | mitigate | Auth.js enforces single-use: token is deleted from `verification_tokens` on first successful consumption. 24h TTL (Auth.js default). Tests in plan 06 (e2e magic-link-login.spec.ts) verify round-trip. |
| T-AUTH-02 | Elevation of Privilege (non-admin signs in) | `src/lib/auth.ts` signIn callback | mitigate | `signIn` returns false unless `admin_user WHERE email=? AND active=true AND role='admin'` exists. Verified by `tests/db/auth-signin-callback.test.ts` (Task 05.3). |
| T-AUTH-03 | Session hijack / stale session | `session: { strategy: 'database', maxAge: 86400 }` + `requireAdmin()` | mitigate | D-09 dual enforcement: (a) 24h idle sliding cap — Auth.js `session.maxAge` + DrizzleAdapter refresh `sessions.expires` each authenticated request; expired sessions return null from `auth()`. Deactivating an admin (`admin_user.active=false`) is handled by the signIn callback — next login attempt fails; live sessions are invalidated by deleting their rows manually. Phase 2 hardens with UX (idle banner + re-auth modal) per CONTEXT.md deferred list; infrastructure enforcement ships in Phase 1. |

A truly stale env var after legitimate admins are added only re-inserts the same email row, which is already present → harmless. |
| T-SEC-ENV | Information Disclosure (secret leak) | `src/lib/auth.ts` | mitigate | `AUTH_SECRET`, `AUTH_RESEND_KEY`, `RESEND_FROM_EMAIL` all in server env; no `NEXT_PUBLIC_*` variant. Verified by `tests/unit/env-validation.test.ts` (plan 01). |

       ```ts
       const schema = z.object({
         email: z.string().email(),
         locale: z.enum(routing.locales),
       });

       export async function requestMagicLink(formData: FormData) {
         const parsed = schema.safeParse({
           email: formData.get('email'),
           locale: formData.get('locale'),
         });
         if (!parsed.success) {
           return { error: 'invalid_input' } as const;
         }
         const { email, locale } = parsed.data;

         try {
           await signIn('resend', {
             email,
             redirect: false,
             redirectTo: `/${locale}/admin`,
           });
           return { ok: true } as const;
         } catch (err) {
           return { error: 'send_failed' } as const;
         }
       }
       ```

    ```ts
    import { describe, it, expect, afterAll } from 'vitest';
    import { eq } from 'drizzle-orm';
    import { db } from '@/db/client';
    import { adminUsers } from '@/db/schema';
    import { requireTestDatabaseUrl } from '../_fixtures/db';

    // Extract the signIn callback by re-running the check inline.
    // Auth.js doesn't expose its config post-construction in a typed way; we replicate
    // the exact query from src/lib/auth.ts here. If the two ever diverge, this test
    // must be updated — that's the point: keep this test in the same review as auth.ts.
    async function signInCheck(email: string | undefined | null): Promise<boolean> {
      if (!email) return false;
      const row = await db
        .select({ role: adminUsers.role, active: adminUsers.active })
        .from(adminUsers)
        .where(eq(adminUsers.email, email))
        .limit(1)
        .then((r) => r[0]);
      return !!row && row.active === true && row.role === 'admin';
    }

    const TEST_ACTIVE   = `test-active-${Date.now()}@example.com`;
    const TEST_INACTIVE = `test-inactive-${Date.now()}@example.com`;

    describe('FOUND-05 / T-AUTH-02: signIn callback authorizes only admin_user.active=true', () => {
      it('rejects email NOT present in admin_user', async () => {
        requireTestDatabaseUrl();
        const result = await signInCheck('ghost@example.com');
        expect(result).toBe(false);
      });

      it('rejects email with no value (null / undefined)', async () => {
        requireTestDatabaseUrl();
        expect(await signInCheck(undefined)).toBe(false);
        expect(await signInCheck(null)).toBe(false);
        expect(await signInCheck('')).toBe(false);
      });

      it('accepts email in admin_user with active=true and role=admin', async () => {
        requireTestDatabaseUrl();
        await db.insert(adminUsers).values({ email: TEST_ACTIVE, role: 'admin', active: true }).onConflictDoNothing();
        const result = await signInCheck(TEST_ACTIVE);
        expect(result).toBe(true);
      });

      it('rejects email in admin_user with active=false', async () => {
        requireTestDatabaseUrl();
        await db.insert(adminUsers).values({ email: TEST_INACTIVE, role: 'admin', active: false }).onConflictDoNothing();
        const result = await signInCheck(TEST_INACTIVE);
        expect(result).toBe(false);
      });

      afterAll(async () => {
        await db.delete(adminUsers).where(eq(adminUsers.email, TEST_ACTIVE));
        await db.delete(adminUsers).where(eq(adminUsers.email, TEST_INACTIVE));
      });
    });
    ```
