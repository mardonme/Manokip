    export async function getProductBySlug(locale: 'uz' | 'ru' | 'en', slug: string): Promise<ProductDetailData | null> {
      'use cache';
      // Step A: resolve product_id from (locale, slug)
      const [resolved] = await db
        .select({ id: productTranslations.productId })
        .from(productTranslations)
        .innerJoin(products, and(eq(products.id, productTranslations.productId), eq(products.status, 'published')))
        .where(and(eq(productTranslations.locale, locale), eq(productTranslations.slug, slug)))
        .limit(1);
      if (!resolved) return null;
      const productId = resolved.id;

      cacheTag(`product:${productId}`);

      // Step B: parallel fetches
      const [productRow, allTranslations, specRows, manufacturerRow, manufacturerTranslationRows, specGroupRows, specGroupTranslationRows, categoryAncestors] = await Promise.all([
        db.select().from(products).where(eq(products.id, productId)).limit(1),
        // Pitfall #6 — fetch ALL 3 locale rows for hreflang
        db.select().from(productTranslations).where(eq(productTranslations.productId, productId)),
        db.select({ /* product_spec_values + JOIN spec_field for key/kind/unit/groupId/sortOrder */ })
          .from(productSpecValues)
          .innerJoin(specFields, eq(specFields.id, productSpecValues.specFieldId))
          .where(eq(productSpecValues.productId, productId)),
        // manufacturer (nullable)
        // manufacturer_translations (all 3)
        // spec_field_groups for the category
        // spec_field_group_translations for current locale
        // category ancestors via recursive CTE OR loop fetching parent_id chain
      ]);

      // ... reshape into ProductDetailData ...
      cacheTag(`category:${productRow[0].categoryId}`);
      if (productRow[0].manufacturerId) cacheTag(`manufacturer:${productRow[0].manufacturerId}`);

      return data;
    }

    export async function generateMetadata({ params }: { params: Promise<{ locale: Locale; slug: string }> }) {
