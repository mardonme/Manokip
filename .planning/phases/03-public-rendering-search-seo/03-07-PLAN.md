    import { sql, and, eq, asc } from 'drizzle-orm';
    import { alias } from 'drizzle-orm/pg-core';
    import { cacheTag } from 'next/cache';
    import { db } from '@/db/client';
    import { manufacturers, manufacturerTranslations, products, productTranslations } from '@/db/schema';

    export type Locale = 'uz' | 'ru' | 'en';

    export interface ManufacturerCardData {
      id: string;
      logoPublicId: string | null;
      name: string;
      slug: string;
      isOfficialRep: boolean;
      productCount: number;
    }

    export async function getManufacturers(locale: Locale): Promise<ManufacturerCardData[]> {
      'use cache';
      cacheTag('manufacturers-list');
      const tCurrent = alias(manufacturerTranslations, 'tCurrent');
      const rows = await db.execute(sql`
        SELECT m.id, m.logo_public_id, m.is_official_rep,
               t.name, t.slug,
               (SELECT COUNT(*) FROM product p WHERE p.manufacturer_id = m.id AND p.status = 'published')::int AS product_count
        FROM manufacturer m
        LEFT JOIN manufacturer_translations t
          ON t.manufacturer_id = m.id AND t.locale = ${locale}::text
        ORDER BY t.name
      `);
      return (rows.rows as any[]).map(r => ({
        id: r.id,
        logoPublicId: r.logo_public_id,
        name: r.name ?? '',
        slug: r.slug ?? '',
        isOfficialRep: r.is_official_rep,
        productCount: r.product_count,
      }));
    }

    export interface ManufacturerDetailData {
      id: string;
      logoPublicId: string | null;
      websiteUrl: string | null;
      isOfficialRep: boolean;
      slugByLocale: { uz?: string; ru?: string; en?: string };
      name: string;
      description: string | null;
      relationshipNote: string | null;
    }

    export async function getManufacturerBySlug(locale: Locale, slug: string): Promise<ManufacturerDetailData | null> {
      'use cache';
      const [resolved] = await db
        .select({ id: manufacturerTranslations.manufacturerId })
        .from(manufacturerTranslations)
        .where(and(eq(manufacturerTranslations.locale, locale), eq(manufacturerTranslations.slug, slug)))
        .limit(1);
      if (!resolved) return null;
      const id = resolved.id;
      cacheTag(`manufacturer:${id}`);
      const [mfg] = await db.select().from(manufacturers).where(eq(manufacturers.id, id)).limit(1);
      if (!mfg) return null;
      const allTranslations = await db.select().from(manufacturerTranslations).where(eq(manufacturerTranslations.manufacturerId, id));
      const slugByLocale: ManufacturerDetailData['slugByLocale'] = {};
      let displayT: typeof allTranslations[number] | undefined = undefined;
      for (const t of allTranslations) {
        slugByLocale[t.locale as Locale] = t.slug;
        if (t.locale === locale) displayT = t;
      }
      // Fallback to uz if current-locale translation missing (cross-locale fallback policy)
      const fallback = displayT ?? allTranslations.find(t => t.locale === 'uz') ?? allTranslations[0];
      return {
        id,
        logoPublicId: mfg.logoPublicId,
        websiteUrl: mfg.websiteUrl,
        isOfficialRep: mfg.isOfficialRep,
        slugByLocale,
        name: fallback?.name ?? '',
        description: fallback?.description ?? null,
        relationshipNote: displayT?.relationshipNote ?? null,
      };
    }

    export async function getManufacturerProducts(manufacturerId: string, locale: Locale, page: number, pageSize: number) {
      'use cache';
      cacheTag(`manufacturer:${manufacturerId}`);
      cacheTag('products-list');
      const offset = (page - 1) * pageSize;
      const [rows, [countRow]] = await Promise.all([
        db.select({
          id: products.id,
          slug: productTranslations.slug,
          name: productTranslations.name,
          shortDesc: productTranslations.shortDesc,
          imagePublicIds: products.imagePublicIds,
          sku: products.sku,
        })
        .from(products)
        .innerJoin(productTranslations,
          and(eq(productTranslations.productId, products.id), eq(productTranslations.locale, locale)))
        .where(and(eq(products.manufacturerId, manufacturerId), eq(products.status, 'published')))
        .orderBy(asc(productTranslations.name))
        .limit(pageSize)
        .offset(offset),
        db.select({ count: sql<number>`count(*)` }).from(products)
          .where(and(eq(products.manufacturerId, manufacturerId), eq(products.status, 'published'))),
      ]);
      return {
        rows: rows.map(r => ({
          id: r.id,
          slug: r.slug,
          name: r.name,
          shortDesc: r.shortDesc,
          heroPublicId: r.imagePublicIds[0] ?? null,
          manufacturerName: null,
          sku: r.sku,
        })),
        total: Number(countRow.count),
      };
    }

    import { setRequestLocale, getTranslations } from 'next-intl/server';
    import { notFound } from 'next/navigation';
    import { CldImage } from 'next-cloudinary';
    import { Badge } from '@/components/ui/badge';
    import { getManufacturerBySlug, getManufacturerProducts } from '@/lib/manufacturer-public';
    import { ProductCard } from '@/components/public/product-card';
    import { buildAlternates, SITE_HOST, type Locale } from '@/lib/metadata';
    import { breadcrumbJsonLd } from '@/lib/jsonld';

    export default async function ManufacturerDetail({ params, searchParams }: {
      params: Promise<{ locale: Locale; slug: string }>;
      searchParams: Promise<{ page?: string }>;
    }) {
      const { locale, slug } = await params;
      setRequestLocale(locale);
      const mfg = await getManufacturerBySlug(locale, slug);
      if (!mfg) notFound();
      const sp = await searchParams;
      const page = Math.max(1, Number(sp.page ?? 1));
      const t = await getTranslations({ locale, namespace: 'public.manufacturer' });
      const { rows, total } = await getManufacturerProducts(mfg.id, locale, page, 24);

      const breadcrumbLd = breadcrumbJsonLd([
        { name: 'Manometr', url: `${SITE_HOST}/${locale}` },
        { name: t('all'), url: `${SITE_HOST}/${locale}/manufacturers` },
        { name: mfg.name, url: `${SITE_HOST}/${locale}/manufacturers/${slug}` },
      ]);

      return (
        <>
          <script type="application/ld+json"
            dangerouslySetInnerHTML={{ __html: JSON.stringify(breadcrumbLd).replace(/</g, '\\u003c') }} />
          <div className="max-w-screen-2xl mx-auto px-6 py-8">
            <div className="flex items-start gap-6 mb-8">
              {mfg.logoPublicId && (
                <CldImage src={mfg.logoPublicId} alt={mfg.name} width={160} height={64} priority />
              )}
              <div className="flex-1">
                <div className="flex items-center gap-2">
                  <h1 className="text-3xl font-semibold text-slate-900">{mfg.name}</h1>
                  {mfg.isOfficialRep && (
                    <Badge variant="default" className="bg-emerald-600 text-white" data-testid="verified-badge">{t('verified')}</Badge>
                  )}
                </div>
                {mfg.relationshipNote && (
                  <p className="text-slate-600 italic mt-2" data-testid="relationship-note">{mfg.relationshipNote}</p>
                )}
                {mfg.description && (
                  <p className="text-slate-700 mt-4 max-w-3xl">{mfg.description}</p>
                )}
              </div>
            </div>
            <h2 className="text-xl font-medium text-slate-900 mb-4" data-testid="products-heading">{t('products', { count: total })}</h2>
            {rows.length > 0 ? (
              <div className="grid grid-cols-1 sm:grid-cols-2 xl:grid-cols-3 gap-6">
                {rows.map(p => <ProductCard key={p.id} product={p} locale={locale} />)}
              </div>
            ) : (
              <p className="text-slate-500">{t('noProducts')}</p>
            )}
          </div>
        </>
      );
    }
