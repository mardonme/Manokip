    ```typescript
    import { test, expect, request } from '@playwright/test';
    const baseURL = process.env.BASE_URL ?? 'http://localhost:3000';

    const PAGES = [
      { name: 'homepage', path: '/uz/' },
      { name: 'category', path: '/uz/categories/manometr' },
      { name: 'product detail', path: '/uz/products/manometr-100' },
      { name: 'manufacturer', path: '/uz/manufacturers/wika' },
    ];

    for (const p of PAGES) {
      test(`SEO-01 + SEO-02: ${p.name} emits hreflang for uz/ru/en + x-default + canonical`, async ({ request }) => {
        const r = await request.get(`${baseURL}${p.path}`);
        expect(r.status()).toBeLessThan(400);
        const html = await r.text();
        // 4 hreflang variants
        expect(html).toMatch(/<link rel="alternate" hreflang="uz"/);
        expect(html).toMatch(/<link rel="alternate" hreflang="ru"/);
        expect(html).toMatch(/<link rel="alternate" hreflang="en"/);
        expect(html).toMatch(/<link rel="alternate" hreflang="x-default"/);
        // canonical present and points to current locale
        expect(html).toMatch(/<link rel="canonical" href="https:\/\/manometr\.uz\/uz/);
      });
    }

    test('CAT-08: product detail emits Product JSON-LD without offers (D-08)', async ({ request }) => {
      const r = await request.get(`${baseURL}/uz/products/manometr-100`);
      const html = await r.text();
      const matches = [...html.matchAll(/<script type="application\/ld\+json">([\s\S]*?)<\/script>/g)];
      const schemas = matches.map(m => JSON.parse(m[1]));
      const product = schemas.find(s => s['@type'] === 'Product');
      expect(product).toBeTruthy();
      expect(product!.offers).toBeUndefined();
      const breadcrumb = schemas.find(s => s['@type'] === 'BreadcrumbList');
      expect(breadcrumb).toBeTruthy();
    });

    test('CAT-08: category listing emits CollectionPage JSON-LD', async ({ request }) => {
      const r = await request.get(`${baseURL}/uz/categories/manometr`);
      const html = await r.text();
      const matches = [...html.matchAll(/<script type="application\/ld\+json">([\s\S]*?)<\/script>/g)];
      const schemas = matches.map(m => JSON.parse(m[1]));
      expect(schemas.find(s => s['@type'] === 'CollectionPage')).toBeTruthy();
    });

    test('SEO-03: /robots.txt + /sitemap-index.xml + /sitemap-uz.xml all return 200 with correct Content-Type', async ({ request }) => {
      const robots = await request.get(`${baseURL}/robots.txt`);
      expect(robots.status()).toBe(200);
      expect(robots.headers()['content-type']).toMatch(/text\/plain/);
      const robotsBody = await robots.text();
      expect(robotsBody).toMatch(/Sitemap:\s*https:\/\/manometr\.uz\/sitemap-index\.xml/);
      const idx = await request.get(`${baseURL}/sitemap-index.xml`);
      expect(idx.status()).toBe(200);
      expect(idx.headers()['content-type']).toMatch(/application\/xml/);
      const idxBody = await idx.text();
      expect(idxBody).toContain('<sitemapindex');
      expect(idxBody).toContain('sitemap-uz.xml');
      const uz = await request.get(`${baseURL}/sitemap-uz.xml`);
      expect(uz.status()).toBe(200);
      const uzBody = await uz.text();
      expect(uzBody).toContain('<urlset');
      expect(uzBody).toMatch(/<xhtml:link rel="alternate" hreflang="uz"/);
    });
    ```

    ```yaml
    name: Lighthouse Preview (SEO-05)

    on:
      pull_request:
        paths:
          - 'src/**'
          - 'next.config.ts'
          - 'package.json'
      workflow_dispatch:

    concurrency:
      group: lighthouse-preview-${{ github.head_ref || github.ref }}
      cancel-in-progress: true

    jobs:
      lighthouse:
        runs-on: ubuntu-latest
        timeout-minutes: 15
        steps:
          - uses: actions/checkout@v4
          - name: Wait for Vercel preview
            id: preview
            uses: patrickedqvist/wait-for-vercel-preview@v1.3.1
            with:
              token: ${{ secrets.GITHUB_TOKEN }}
              max_timeout: 600
          - name: Run Lighthouse CI
            uses: treosh/lighthouse-ci-action@v12
            with:
              urls: |
                ${{ steps.preview.outputs.url }}/uz/products/manometr-100
              configPath: ./.lighthouserc.json
              uploadArtifacts: true
              temporaryPublicStorage: true
            env:
              # Vercel Deployment Protection bypass — same secret used by e2e-preview workflow
    ```
