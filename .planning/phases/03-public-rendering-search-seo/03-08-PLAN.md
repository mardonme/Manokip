    ```typescript
    import { sql, eq, and } from 'drizzle-orm';
    import { cacheTag } from 'next/cache';
    import { db } from '@/db/client';
    import { products, productTranslations, categories, categoryTranslations, manufacturers, manufacturerTranslations } from '@/db/schema';

    export type Locale = 'uz' | 'ru' | 'en';
    const HOST = 'https://manometr.uz';
    const ALL_LOCALES: Locale[] = ['uz', 'ru', 'en'];

    export interface SitemapEntry {
      loc: string;                     // absolute URL for current locale
      lastmod?: string;                // ISO timestamp
      alternates: Partial<Record<Locale, string>>; // for xhtml:link
    }

    export async function buildLocaleSitemapEntries(locale: Locale): Promise<SitemapEntry[]> {
      'use cache';
      cacheTag('sitemap');

      const entries: SitemapEntry[] = [];

      // Static paths
      entries.push(
        {
          loc: `${HOST}/${locale}`,
          alternates: Object.fromEntries(ALL_LOCALES.map(l => [l, `${HOST}/${l}`])) as any,
        },
        {
          loc: `${HOST}/${locale}/categories`,
          alternates: Object.fromEntries(ALL_LOCALES.map(l => [l, `${HOST}/${l}/categories`])) as any,
        },
        {
          loc: `${HOST}/${locale}/manufacturers`,
          alternates: Object.fromEntries(ALL_LOCALES.map(l => [l, `${HOST}/${l}/manufacturers`])) as any,
        },
      );

      // Products: SELECT all 3 locale slugs grouped by product_id
      const productRows = await db.execute(sql`
        SELECT p.id, p.updated_at,
               MAX(CASE WHEN t.locale='uz' THEN t.slug END) AS slug_uz,
               MAX(CASE WHEN t.locale='ru' THEN t.slug END) AS slug_ru,
               MAX(CASE WHEN t.locale='en' THEN t.slug END) AS slug_en
        FROM product p
        JOIN product_translations t ON t.product_id = p.id
        WHERE p.status = 'published'
        GROUP BY p.id
      `);
      for (const r of productRows.rows as any[]) {
        const slugForLocale = r[`slug_${locale}`];
        if (!slugForLocale) continue; // skip if current-locale translation missing
        const alternates: Partial<Record<Locale, string>> = {};
        for (const l of ALL_LOCALES) {
          const s = r[`slug_${l}`];
          if (s) alternates[l] = `${HOST}/${l}/products/${s}`;
        }
        entries.push({
          loc: `${HOST}/${locale}/products/${slugForLocale}`,
          lastmod: r.updated_at?.toISOString?.() ?? new Date(r.updated_at).toISOString(),
          alternates,
        });
      }

      // Categories: same shape
      const categoryRows = await db.execute(sql`
        SELECT c.id, c.updated_at,
               MAX(CASE WHEN ct.locale='uz' THEN ct.slug END) AS slug_uz,
               MAX(CASE WHEN ct.locale='ru' THEN ct.slug END) AS slug_ru,
               MAX(CASE WHEN ct.locale='en' THEN ct.slug END) AS slug_en
        FROM category c
        JOIN category_translations ct ON ct.category_id = c.id
        GROUP BY c.id
      `);
      for (const r of categoryRows.rows as any[]) {
        const slugForLocale = r[`slug_${locale}`];
        if (!slugForLocale) continue;
        const alternates: Partial<Record<Locale, string>> = {};
        for (const l of ALL_LOCALES) {
          const s = r[`slug_${l}`];
          if (s) alternates[l] = `${HOST}/${l}/categories/${s}`;
        }
        entries.push({
          loc: `${HOST}/${locale}/categories/${slugForLocale}`,
          lastmod: r.updated_at?.toISOString?.() ?? new Date(r.updated_at).toISOString(),
          alternates,
        });
      }

      // Manufacturers: same shape
      const mfgRows = await db.execute(sql`
        SELECT m.id, m.updated_at,
               MAX(CASE WHEN mt.locale='uz' THEN mt.slug END) AS slug_uz,
               MAX(CASE WHEN mt.locale='ru' THEN mt.slug END) AS slug_ru,
               MAX(CASE WHEN mt.locale='en' THEN mt.slug END) AS slug_en
        FROM manufacturer m
        JOIN manufacturer_translations mt ON mt.manufacturer_id = m.id
        GROUP BY m.id
      `);
      for (const r of mfgRows.rows as any[]) {
        const slugForLocale = r[`slug_${locale}`];
        if (!slugForLocale) continue;
        const alternates: Partial<Record<Locale, string>> = {};
        for (const l of ALL_LOCALES) {
          const s = r[`slug_${l}`];
          if (s) alternates[l] = `${HOST}/${l}/manufacturers/${s}`;
        }
        entries.push({
          loc: `${HOST}/${locale}/manufacturers/${slugForLocale}`,
          lastmod: r.updated_at?.toISOString?.() ?? new Date(r.updated_at).toISOString(),
          alternates,
        });
      }

      return entries;
    }

    export function escapeXml(s: string): string {
      return s.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;').replace(/'/g, '&apos;');
    }

    export function renderUrlsetXml(entries: SitemapEntry[]): string {
      const lines: string[] = [];
      lines.push('<?xml version="1.0" encoding="UTF-8"?>');
      lines.push('<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9" xmlns:xhtml="http://www.w3.org/1999/xhtml">');
      for (const e of entries) {
        lines.push('  <url>');
        lines.push(`    <loc>${escapeXml(e.loc)}</loc>`);
        if (e.lastmod) lines.push(`    <lastmod>${e.lastmod}</lastmod>`);
        for (const [l, url] of Object.entries(e.alternates)) {
          if (url) lines.push(`    <xhtml:link rel="alternate" hreflang="${l}" href="${escapeXml(url)}"/>`);
        }
        // x-default points to uz variant
        if (e.alternates.uz) lines.push(`    <xhtml:link rel="alternate" hreflang="x-default" href="${escapeXml(e.alternates.uz)}"/>`);
        lines.push('  </url>');
      }
      lines.push('</urlset>');
      return lines.join('\n');
    }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-08-01 | XSS-via-XML | slug containing XML special chars (`<`, `&`) breaking sitemap parsing | mitigate | escapeXml() applied to every dynamic value before <loc>/<xhtml:link href> emission; slugs are also normalized via `src/lib/slug.ts` before write so `<` and `&` should not appear in practice — defense in depth |
| T-03-08-02 | Information Disclosure | sitemap exposes draft product slugs | mitigate | SQL filter `WHERE p.status = 'published'` (already in buildLocaleSitemapEntries) — drafts excluded |
| T-03-08-03 | DoS | crawler hammering sitemap with no cache | mitigate | 'use cache' + cacheTag('sitemap') means cold rebuild only happens when revalidateTag fires (admin write path); steady-state is served from edge cache with sub-100ms latency |
</threat_model>
