    export interface EnumFacet { value: string; count: number; }
    export async function getEnumFacetCounts(specFieldKey: string, categoryId: string): Promise<EnumFacet[]> {
      'use cache';
      cacheTag(`category:${categoryId}`);
      cacheTag('products-list');
      const rows = await db.execute(sql`
        SELECT v.enum_value AS value, COUNT(DISTINCT v.product_id)::int AS count
        FROM product_spec_values v
        JOIN spec_field sf ON sf.id = v.spec_field_id
        JOIN product p ON p.id = v.product_id
        WHERE sf.key = ${specFieldKey}
          AND p.category_id = ${categoryId}::uuid
          AND p.status = 'published'
          AND v.enum_value IS NOT NULL
        GROUP BY v.enum_value
        ORDER BY count DESC
      `);

      params: Promise<{ locale: Locale; slug: string[] }>;
      searchParams: Promise<Record<string, string | string[] | undefined>>;
    }) {
      const { locale, slug } = await params;
      setRequestLocale(locale);
      const slugStr = slug.join('/');
      const category = await getCategoryBySlug(locale, slugStr);
      if (!category) notFound();

      // Pull spec-field schema FIRST so we know which dynamic filter keys to parse.
      // Labels arrive per-locale from spec_field_translations join.
      const filterSchema = await getCategoryFilterSchema(category.id, locale);

      // Build a dynamic filter cache from filterSchema + parse OUTSIDE 'use cache'
      const sp = await searchParams;
      const base = await baseFilterCache.parse(sp);
      const dynamicFilters: CategoryFilterValue[] = parseDynamicFilters(filterSchema, sp);

      // Cached data fetches
      const [{ rows, total }, facetData] = await Promise.all([
        getCategoryProducts(category.id, locale, dynamicFilters, base.page, base.pageSize),
        loadFacetDataForSchema(filterSchema, category.id),
      ]);

      const breadcrumbs = [
        { name: 'Manometr', url: `${SITE_HOST}/${locale}` },
        { name: t('catalog'), url: `${SITE_HOST}/${locale}/categories` },
        { name: category.nameByLocale[locale], url: `${SITE_HOST}/${locale}/categories/${slugStr}` },
      ];
      const collectionLd = collectionPageJsonLd(category.nameByLocale[locale], rows.map(r => `${SITE_HOST}/${locale}/products/${r.slug}`));
      const breadcrumbLd = breadcrumbJsonLd(breadcrumbs);

      return (
        <>
          <script type="application/ld+json"
            dangerouslySetInnerHTML={{ __html: JSON.stringify(collectionLd).replace(/</g, '\\u003c') }} />

            <div>
              <ActiveFilterPills filterSchema={filterSchema} />
              <div className="text-sm text-slate-500 mb-3" data-testid="results-count">{t('resultsCount', { count: total })}</div>
              {rows.length === 0 ? (
                <p>{t('noResults')}</p>
              ) : (
                <div className="grid grid-cols-1 sm:grid-cols-2 xl:grid-cols-3 gap-6">
                  {rows.map(p => <ProductCard key={p.id} product={p} locale={locale} />)}
                </div>
              )}
              {/* simple page links: prev / next based on total */}
            </div>
